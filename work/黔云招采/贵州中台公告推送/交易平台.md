POST /qyzc/center/result

```json
{
  "clientId": "string",
  "type": "string",
  "dataSetCode": "string",
  "mediaSetCode": "string",
  "backTime": "string",
  "data": {
    "Success": true,
    "Code": "string",
    "Message": "string",
    "DataType": "string",
    "TypeCode": "string",
    "TypeName": "string",
    "ReturnCode": "string",
    "AuditState": 0,
    "AuditStateName": "string",
    "AuditReason": "string"
  },
  "reqTime": "string",
  "requestId": "string"
}
```

```json
{
  "code": "string",
  "Message": "string",
  "resTime": "string"
}

```
